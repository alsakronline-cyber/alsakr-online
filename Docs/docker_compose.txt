version: '3.8'

services:
  # FastAPI Backend
  backend:
    build:
      context: ./backend
      dockerfile: Dockerfile
    container_name: nexus-backend
    restart: unless-stopped
    ports:
      - "8000:8000"
    environment:
      - DATABASE_URL=sqlite:////app/data/nexus.db
      - OLLAMA_HOST=http://ollama:11434
      - QDRANT_HOST=http://qdrant:6333
      - POCKETBASE_URL=http://pocketbase:8090
      - JWT_SECRET=${JWT_SECRET}
      - SMTP_SERVER=smtp.gmail.com
      - SMTP_PORT=587
      - EMAIL_USERNAME=${EMAIL_USERNAME}
      - EMAIL_PASSWORD=${EMAIL_PASSWORD}
    volumes:
      - ./backend/data:/app/data
      - ./backend/logs:/app/logs
    depends_on:
      - ollama
      - qdrant
      - pocketbase
    networks:
      - nexus-network

  # Ollama AI Server
  ollama:
    image: ollama/ollama:latest
    container_name: nexus-ollama
    restart: unless-stopped
    ports:
      - "11434:11434"
    volumes:
      - ./data/ollama:/root/.ollama
    networks:
      - nexus-network

  # Qdrant Vector Database
  qdrant:
    image: qdrant/qdrant:latest
    container_name: nexus-qdrant
    restart: unless-stopped
    ports:
      - "6333:6333"
    volumes:
      - ./data/qdrant:/qdrant/storage
    networks:
      - nexus-network

  # PocketBase CRM
  pocketbase:
    image: ghcr.io/muchobien/pocketbase:latest
    container_name: nexus-pocketbase
    restart: unless-stopped
    ports:
      - "8090:8090"
    volumes:
      - ./data/pocketbase:/pb_data
    networks:
      - nexus-network

  # n8n Workflow Automation
  n8n:
    image: n8nio/n8n:latest
    container_name: nexus-n8n
    restart: unless-stopped
    ports:
      - "5678:5678"
    environment:
      - N8N_BASIC_AUTH_ACTIVE=true
      - N8N_BASIC_AUTH_USER=admin
      - N8N_BASIC_AUTH_PASSWORD=${N8N_PASSWORD}
      - WEBHOOK_URL=https://nexusindustrial.com
    volumes:
      - ./data/n8n:/home/node/.n8n
    networks:
      - nexus-network

  # Caddy Reverse Proxy
  caddy:
    image: caddy:2-alpine
    container_name: nexus-caddy
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./caddy/Caddyfile:/etc/caddy/Caddyfile
      - ./data/caddy/data:/data
      - ./data/caddy/config:/config
    networks:
      - nexus-network

networks:
  nexus-network:
    driver: bridge

volumes:
  ollama_data:
  qdrant_data:
  pocketbase_data:
  n8n_data:
  caddy_data:
  caddy_config:
